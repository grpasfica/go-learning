data=(3 6 7 8 9 11 4 9 5 2 2 4 6 6)
total_data=${#data[@]}
echo $total_data
echo "Total data : $total_data"

echo "Bilangan genap terdiri dari : "

count=0
jml_bil=0
while [ $count -lt $total_data ]; do
  if ((data[count] % 2 == 0)); then
    (( jml_bil += data[count]))
    echo "${data[count]}"
  fi
    ((count++))
done

echo "Jumlah seluruh bilangan genap : ${jml_bil}"
